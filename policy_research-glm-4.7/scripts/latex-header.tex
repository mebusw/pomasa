% LaTeX头部配置文件
% 用于控制PDF输出格式，特别是中文支持

% 字体配置 - 中文支持

\usepackage{xeCJK}

\setCJKmainfont{Songti SC}

    linkcolor=blue,
    urlcolor=blue,
    citecolor=blue
}

\usepackage{graphicx}
